import { SaveOptions } from 'mongoose';
import {
	createSchema,
	ExtractDoc,
	ExtractProps,
	typedModel
} from 'ts-mongoose';

export const ${MODEL}Schema = createSchema({}, {});

export const ${MODEL} = typedModel('${MODEL}', ${MODEL}Schema, '${COLLECTION}', undefined, {
	create: (props: Partial<${MODEL}Props>): ${MODEL}Doc => new ${MODEL}(props),
	createAndSave: (
		props: Partial<${MODEL}Props>,
		options: SaveOptions = {}
	): Promise<${MODEL}Doc> => new ${MODEL}(props).save(options)
});

export type ${MODEL}Doc = ExtractDoc<typeof ${MODEL}Schema>;
export type ${MODEL}Props = ExtractProps<typeof ${MODEL}Schema>;
